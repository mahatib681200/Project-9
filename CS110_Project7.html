# prompt: create a vertical bar chart with the count of teslas for each city

import matplotlib.pyplot as plt
# Filter for Tesla vehicles
tesla_df = df[df['Make'] == 'TESLA']

# Count the number of Teslas in each city
tesla_city_counts = tesla_df['City'].value_counts().reset_index()
tesla_city_counts.columns = ['City', 'Count']

# Sort by count for better visualization
tesla_city_counts = tesla_city_counts.sort_values(by='Count', ascending=False)

# Take the top N cities to avoid clutter, adjust N as needed
top_n_cities = 20
tesla_city_counts_top = tesla_city_counts.head(top_n_cities)

# Create a vertical bar chart
plt.figure(figsize=(12, 8))
sns.barplot(x='City', y='Count', data=tesla_city_counts_top, palette='viridis')
plt.title(f'Count of Tesla Vehicles per City (Top {top_n_cities} Cities)', fontsize=16)
plt.xlabel('City', fontsize=12)
plt.ylabel('Number of Teslas', fontsize=12)

plt.tight_layout() # Adjust layout to prevent labels overlapping
plt.show()

# prompt: create a line chart with the count of luxury cars per city

import matplotlib.pyplot as plt
# Group by city and make, then count the occurrences
car_counts_by_city_make = df.groupby(['City', 'Make']).size().reset_index(name='Count')

# Define a list of common luxury car makes (you might need to adjust this based on your data)
luxury_makes = ['TESLA', 'BMW', 'AUDI', 'MERCEDES-BENZ', 'PORSCHE', 'JAGUAR', 'LEXUS', 'CADILLAC']

# Filter the DataFrame to include only luxury makes
luxury_car_counts = car_counts_by_city_make[car_counts_by_city_make['Make'].isin(luxury_makes)]

# Group by city and sum the counts of luxury cars
luxury_cars_per_city = luxury_car_counts.groupby('City')['Count'].sum().reset_index()

# Sort by count for better visualization
luxury_cars_per_city = luxury_cars_per_city.sort_values(by='Count', ascending=False)

# Take the top N cities to avoid clutter, adjust N as needed
top_n_cities = 20
luxury_cars_per_city_top = luxury_cars_per_city.head(top_n_cities)

# Create a line chart
plt.figure(figsize=(14, 7))
sns.lineplot(x='City', y='Count', data=luxury_cars_per_city_top, marker='o')
plt.title(f'Total Count of Luxury Vehicles per City (Top {top_n_cities} Cities)', fontsize=16)
plt.xlabel('City', fontsize=12)
plt.ylabel('Number of Luxury Cars', fontsize=12)

plt.grid(True) # Add a grid for better readability
